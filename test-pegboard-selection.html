Positioning & Scaling:

positionPegboardModel(model, dimensions) { // 1. Calculate bounding box const box = new THREE.Box3().setFromObject(model); const size = new THREE.Vector3(); box.getSize(size); // 2. Center model at origin const center = new THREE.Vector3(); box.getCenter(center); model.position.sub(center); // 3. Scale to match target dimensions const scaleX = dimensions.width / size.x; const scaleY = dimensions.height / size.y; const uniformScale = Math.min(scaleX, scaleY); model.scale.set(uniformScale, uniformScale, uniformScale); // 4. Position above ground model.position.y = 0; }
